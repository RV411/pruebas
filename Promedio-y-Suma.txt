#include <stdio.h>


int main(void)
{

    int num = 1;
    int res = 0;
    int cont = 0;
    int suma = 0;
    int suma2 = 0;
    float prom = 0;
    while (num > 0)
    {

        scanf("%d", &num);

        if (num == 0)
        {
            printf("%d\n", suma);
            prom = suma2 / cont;
            printf("%.2f\n", prom);
        }


        res = num % 2;

        {

            suma = suma + num;

        }
    }
}
