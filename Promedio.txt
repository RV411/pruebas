#include <stdio.h>

int main(void)
{


    float num = 1;
    float suma = 0;
    int cont = 0;
    float prom = 0.0f;
    while (num > 0)
    {

        scanf("%f", &num);
        suma = suma + num;

        if (num == 0)
        {
            prom = suma / cont;
            printf(" %.2f", prom);
            return 0;
        }


        cont = cont + 1;
    }
}
